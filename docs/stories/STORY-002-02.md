## Overview

This story implements JSON-based data persistence for saving and loading student progress, with auto-save functionality and corruption handling.

## User Story

As a student, I want my progress to be saved automatically so that I don't lose my work if the game closes unexpectedly.

## Acceptance Criteria

### Functional Requirements

- [ ] Auto-save after each word completion
- [ ] Store progress in `data/progress/{student_id}_progress.json`
- [ ] Load existing data on student profile selection
- [ ] Handle missing files gracefully (create new)
- [ ] Handle corrupt files gracefully (backup + recover what possible)
- [ ] Create backup before each save operation
- [ ] Validate JSON structure on load

### Non-Functional Requirements

- [ ] Performance: Save operation completes in < 500ms
- [ ] Accessibility: N/A (background process)
- [ ] Error handling: Queue saves if disk full, retry later

```python
# src/components/data_store.py
import json
import os
import shutil
from datetime import datetime
from typing import Any, Optional

class DataStore:
    def __init__(self, base_path: str = "data/progress"):
        self.base_path = base_path
        os.makedirs(base_path, exist_ok=True)
        
    def _get_file_path(self, student_id: str) -> str:
        return os.path.join(self.base_path, f"{student_id}_progress.json")
    
    def _create_backup(self, file_path: str) -> Optional[str]:
        """Create timestamped backup, return backup path"""
        if not os.path.exists(file_path):
            return None
        backup_dir = os.path.join(os.path.dirname(file_path), "backup")
        os.makedirs(backup_dir, exist_ok=True)
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_path = os.path.join(backup_dir, f"backup_{timestamp}.json")
        shutil.copy2(file_path, backup_path)
        return backup_path
    
    def save(self, student_id: str, data: dict) -> bool:
        """Save data with backup"""
        file_path = self._get_file_path(student_id)
        try:
            # Create backup first
            self._create_backup(file_path)
            # Save new data
            with open(file_path, 'w') as f:
                json.dump(data, f, indent=2)
            return True
        except Exception as e:
            print(f"Save failed: {e}")
            return False
    
    def load(self, student_id: str) -> Optional[dict]:
        """Load data with validation"""
        file_path = self._get_file_path(student_id)
        if not os.path.exists(file_path):
            return self._create_empty_progress(student_id)
        
        try:
            with open(file_path, 'r') as f:
                data = json.load(f)
            if self._validate_schema(data):
                return data
            else:
                return self._recover_from_corrupt(file_path)
        except json.JSONDecodeError:
            return self._recover_from_corrupt(file_path)
    
    def _validate_schema(self, data: dict) -> bool:
        """Check if data has required fields"""
        required = ['student_id', 'sessions', 'mastered_words']
        return all(field in data for field in required)
    
    def _create_empty_progress(self, student_id: str) -> dict:
        """Create new empty progress structure"""
        return {
            'version': '1.0',
            'student_id': student_id,
            'created_at': datetime.now().isoformat(),
            'sessions': [],
            'mastered_words': [],
            'needs_practice': [],
            'achievements': []
        }
```
