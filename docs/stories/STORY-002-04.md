## Overview

This story implements accuracy rate calculation with trend indicators that show whether performance is improving, stable, or declining compared to previous sessions.

## User Story

As a parent and student, I want to see my accuracy rate and whether it's improving so that we can understand learning progress.

## Acceptance Criteria

### Functional Requirements

- [ ] Calculate percentage accuracy (correct attempts / total attempts)
- [ ] Show trend arrow (↑ improving / → stable / ↓ declining)
- [ ] Compare to previous session accuracy
- [ ] Compare to weekly average accuracy
- [ ] Display accuracy as percentage (e.g., "85%")
- [ ] Update accuracy after each session

### Non-Functional Requirements

- [ ] Performance: Calculation completes in < 50ms
- [ ] Accessibility: Trend communicated visually AND textually (not just arrow color)
- [ ] Error handling: Handle insufficient data gracefully ("No previous data")

```python
# src/components/progress_tracker.py
class ProgressTracker:
    # ... existing code ...
    
    def calculate_accuracy(self, session_id: str = None) -> float:
        """Calculate accuracy percentage"""
        sessions = self.progress_data.get('sessions', [])
        if not sessions:
            return 0.0
        
        if session_id:
            sessions = [s for s in sessions if s['session_id'] == session_id]
        
        total_words = sum(len(s.get('words', [])) for s in sessions)
        if total_words == 0:
            return 0.0
        
        correct_words = sum(
            sum(1 for w in s.get('words', []) if w.get('correct'))
            for s in sessions
        )
        return (correct_words / total_words) * 100
    
    def get_accuracy_trend(self) -> str:
        """Determine trend compared to previous period"""
        sessions = sorted(
            self.progress_data.get('sessions', []),
            key=lambda s: s.get('date', '')
        )
        
        if len(sessions) < 2:
            return "new"  # No comparison possible
        
        # Compare last session to previous session
        last_accuracy = self.calculate_accuracy(sessions[-1]['session_id'])
        prev_accuracy = self.calculate_accuracy(sessions[-2]['session_id'])
        
        diff = last_accuracy - prev_accuracy
        
        if diff > 5:
            return "improving"
        elif diff < -5:
            return "declining"
        else:
            return "stable"
    
    def get_accuracy_display(self) -> dict:
        """Get formatted accuracy display data"""
        accuracy = self.calculate_accuracy()
        trend = self.get_accuracy_trend()
        
        trend_symbols = {
            "improving": ("↑", "Improving"),
            "stable": ("→", "Stable"),
            "declining": ("↓", "Needs Practice"),
            "new": ("-", "New")
        }
        
        symbol, label = trend_symbols.get(trend, ("?", "Unknown"))
        
        return {
            "percentage": round(accuracy, 1),
            "trend_symbol": symbol,
            "trend_label": label
        }
```
