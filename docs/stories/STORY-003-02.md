```python
# student_profile.py
from dataclasses import dataclass
from datetime import datetime
from typing import Optional

@dataclass
class StudentProfile:
    id: str  # UUID or timestamp-based unique ID
    name: str
    avatar_id: str  # Identifier for selected avatar image
    created_date: datetime
    last_played: Optional[datetime] = None
    difficulty_level: str = "medium"  # beginner, medium, advanced
    
    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "name": self.name,
            "avatar_id": self.avatar_id,
            "created_date": self.created_date.isoformat(),
            "last_played": self.last_played.isoformat() if self.last_played else None,
            "difficulty_level": self.difficulty_level
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> "StudentProfile":
        return cls(
            id=data["id"],
            name=data["name"],
            avatar_id=data["avatar_id"],
            created_date=datetime.fromisoformat(data["created_date"]),
            last_played=datetime.fromisoformat(data["last_played"]) if data.get("last_played") else None,
            difficulty_level=data.get("difficulty_level", "medium")
        )

# profile_manager.py
import json
import os
import uuid
from typing import List, Optional

class ProfileManager:
    AVATAR_OPTIONS = [
        "astronaut", "alien", "rocket", "planet", 
        "star", "moon", "robot", "cat"
    ]
    
    def __init__(self, data_path: str):
        self.data_path = data_path
        self._ensure_data_exists()
    
    def _ensure_data_exists(self):
        """Create default profiles file if doesn't exist."""
        if not os.path.exists(self.data_path):
            default_data = {"profiles": []}
            self._save_data(default_data)
    
    def _load_data(self) -> dict:
        """Load profiles from JSON file."""
        with open(self.data_path, "r") as f:
            return json.load(f)
    
    def _save_data(self, data: dict):
        """Save profiles to JSON file."""
        with open(self.data_path, "w") as f:
            json.dump(data, f, indent=2)
    
    def create_profile(self, name: str, avatar_id: str) -> StudentProfile:
        """Create a new student profile."""
        if not self._is_valid_name(name):
            raise ValueError("Invalid name")
        if self._name_exists(name):
            raise ValueError("Profile with this name already exists")
        
        profile = StudentProfile(
            id=str(uuid.uuid4()),
            name=name,
            avatar_id=avatar_id,
            created_date=datetime.now()
        )
        data = self._load_data()
        data["profiles"].append(profile.to_dict())
        self._save_data(data)
        return profile
    
    def get_profile(self, profile_id: str) -> Optional[StudentProfile]:
        """Get a profile by ID."""
        data = self._load_data()
        for p in data["profiles"]:
            if p["id"] == profile_id:
                return StudentProfile.from_dict(p)
        return None
    
    def get_all_profiles(self) -> List[StudentProfile]:
        """Get all student profiles."""
        data = self._load_data()
        return [StudentProfile.from_dict(p) for p in data["profiles"]]
    
    def update_profile(self, profile_id: str, name: str = None, avatar_id: str = None) -> StudentProfile:
        """Update an existing profile."""
        data = self._load_data()
        for i, p in enumerate(data["profiles"]):
            if p["id"] == profile_id:
                if name:
                    p["name"] = name
                if avatar_id:
                    p["avatar_id"] = avatar_id
                data["profiles"][i] = p
                self._save_data(data)
                return StudentProfile.from_dict(p)
        raise ValueError("Profile not found")
    
    def delete_profile(self, profile_id: str, has_progress: bool = False) -> bool:
        """Delete a profile (with confirmation if has progress)."""
        if has_progress:
            raise ValueError("Cannot delete profile with progress without confirmation")
        data = self._load_data()
        data["profiles"] = [p for p in data["profiles"] if p["id"] != profile_id]
        self._save_data(data)
        return True
    
    def _is_valid_name(self, name: str) -> bool:
        """Validate profile name."""
        return 1 <= len(name.strip()) <= 30
    
    def _name_exists(self, name: str) -> bool:
        """Check if name already exists."""
        return any(p["name"].lower() == name.lower() for p in self._load_data()["profiles"])
```
