## Overview

Implement full CRUD (Create, Read, Update, Delete) operations for custom spelling word lists, including adding, editing, and removing individual words with difficulty tags.

## User Story

As a parent, I want to create and manage custom word lists so that I can include my child's school spelling words or custom vocabulary.

## Acceptance Criteria

### Functional Requirements

- [ ] View all words in a student's word list
- [ ] Add new word with definition/context
- [ ] Edit existing word (spelling, definition, difficulty tag)
- [ ] Delete individual words (with confirmation)
- [ ] Tag words by difficulty (beginner, medium, advanced)
- [ ] Bulk add words (paste multiple words, one per line)
- [ ] Word list displays word count and difficulty breakdown
- [ ] Changes persist immediately to storage
- [ ] Search/filter words by text or difficulty

### Non-Functional Requirements

- [ ] Performance: List loads < 300ms for 100+ words
- [ ] Accessibility: All operations keyboard accessible
- [ ] Error handling: Validate word format, handle duplicates gracefully
- [ ] Storage: Efficient JSON storage with proper indexing

# word_entry.py
from dataclasses import dataclass
from datetime import datetime
from typing import Optional
from enum import Enum

class Difficulty(Enum):
    BEGINNER = "beginner"
    MEDIUM = "medium"
    ADVANCED = "advanced"

@dataclass
class WordEntry:
    id: str  # UUID
    spelling: str
    definition: str
    difficulty: Difficulty
    created_date: datetime
    last_modified: datetime
    
    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "spelling": self.spelling,
            "definition": self.definition,
            "difficulty": self.difficulty.value,
            "created_date": self.created_date.isoformat(),
            "last_modified": self.last_modified.isoformat()
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> "WordEntry":
        return cls(
            id=data["id"],
            spelling=data["spelling"],
            definition=data["definition"],
            difficulty=Difficulty(data["difficulty"]),
            created_date=datetime.fromisoformat(data["created_date"]),
            last_modified=datetime.fromisoformat(data["last_modified"])
        )

# word_list_manager.py
import json
import os
import uuid
from typing import List, Optional
from src.models.word_entry import WordEntry, Difficulty

class WordListManager:
    def __init__(self, data_dir: str):
        self.data_dir = data_dir
        os.makedirs(data_dir, exist_ok=True)
    
    def _get_profile_path(self, profile_id: str) -> str:
        return os.path.join(self.data_dir, f"{profile_id}.json")
    
    def _load_data(self, profile_id: str) -> dict:
        path = self._get_profile_path(profile_id)
        if not os.path.exists(path):
            return {"words": [], "meta": {"created": None, "updated": None}}
        with open(path, "r") as f:
            return json.load(f)
    
    def _save_data(self, profile_id: str, data: dict):
        path = self._get_profile_path(profile_id)
        data["meta"]["updated"] = datetime.now().isoformat()
        with open(path, "w") as f:
            json.dump(data, f, indent=2)
    
    def add_word(self, profile_id: str, spelling: str, definition: str = "", 
                 difficulty: Difficulty = Difficulty.MEDIUM) -> WordEntry:
        """Add a new word to the list."""
        if not self._is_valid_spelling(spelling):
            raise ValueError("Invalid spelling")
        
        word = WordEntry(
            id=str(uuid.uuid4()),
            spelling=spelling.upper(),
            definition=definition,
            difficulty=difficulty,
            created_date=datetime.now(),
            last_modified=datetime.now()
        )
        
        data = self._load_data(profile_id)
        data["words"].append(word.to_dict())
        self._save_data(profile_id, data)
        return word
    
    def get_words(self, profile_id: str, 
                  difficulty: Optional[Difficulty] = None,
                  search_terms: Optional[str] = None) -> List[WordEntry]:
        """Get words with optional filtering."""
        data = self._load_data(profile_id)
        words = [WordEntry.from_dict(w) for w in data["words"]]
        
        if difficulty:
            words = [w for w in words if w.difficulty == difficulty]
        
        if search_terms:
            terms = search_terms.lower().split()
            words = [w for w in words if 
                    any(term in w.spelling.lower() or term in w.definition.lower() 
                      for term in terms)]
        
        return words
    
    def update_word(self, profile_id: str, word_id: str, 
                    spelling: str = None, definition: str = None,
                    difficulty: Difficulty = None) -> WordEntry:
        """Update an existing word."""
        data = self._load_data(profile_id)
        for i, w in enumerate(data["words"]):
            if w["id"] == word_id:
                if spelling:
                    w["spelling"] = spelling.upper()
                if definition is not None:
                    w["definition"] = definition
                if difficulty:
                    w["difficulty"] = difficulty.value
                w["last_modified"] = datetime.now().isoformat()
                data["words"][i] = w
                self._save_data(profile_id, data)
                return WordEntry.from_dict(w)
        raise ValueError("Word not found")
    
    def delete_word(self, profile_id: str, word_id: str) -> bool:
        """Delete a word from the list."""
        data = self._load_data(profile_id)
        initial_len = len(data["words"])
        data["words"] = [w for w in data["words"] if w["id"] != word_id]
        if len(data["words"]) < initial_len:
            self._save_data(profile_id, data)
            return True
        raise ValueError("Word not found")
    
    def add_bulk_words(self, profile_id: str, text: str, 
                       difficulty: Difficulty = Difficulty.MEDIUM) -> List[WordEntry]:
        """Add multiple words from newline-separated text."""
        added = []
        for line in text.strip().split("\n"):
            spelling = line.strip().split()[0] if line.strip() else ""
            if spelling and self._is_valid_spelling(spelling):
                # Try to extract definition after first space or tab
                parts = line.strip().split(maxsplit=1)
                definition = parts[1] if len(parts) > 1 else ""
                try:
                    word = self.add_word(profile_id, spelling, definition, difficulty)
                    added.append(word)
                except ValueError:
                    continue  # Skip invalid words
        return added
    
    def _is_valid_spelling(self, spelling: str) -> bool:
        """Validate word spelling."""
        if not spelling or not spelling.strip():
            return False
        if len(spelling.strip()) < 2 or len(spelling.strip()) > 25:
            return False
        if not spelling.strip().replace("-", "").isalpha():
            return False
        return True
    
    def get_word_count(self, profile_id: str) -> dict:
        """Get word count statistics."""
        words = self.get_words(profile_id)
        return {
            "total": len(words),
            "beginner": len([w for w in words if w.difficulty == Difficulty.BEGINNER]),
            "medium": len([w for w in words if w.difficulty == Difficulty.MEDIUM]),
            "advanced": len([w for w in words if w.difficulty == Difficulty.ADVANCED])
        }

## Notes & Considerations

- **Default Words**: Schools may have pre-built lists to import (see STORY-003-04 for CSV)
- **Word Validation**: Should we allow hyphenated words? Yes (handled). Numbers? No.
- **Case Handling**: Store as UPPERCASE for consistent gameplay, display as entered
- **Definition Length**: Consider max length (suggested 255 chars)
- **Sort Order**: Default to alphabetical, allow sorting by difficulty

### Implementation Tips

- Use debounced search for performance with large lists
- Consider limiting list to 200 words for MVP (pagination for future)
- Store word lists in per-profile JSON files for easy association
- Add visual feedback when saving (spinner or brief "Saved" message)
