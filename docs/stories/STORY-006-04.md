## Overview

Implement full keyboard navigation support throughout the game, ensuring all interactive elements can be accessed and controlled without a mouse. This enables students with motor disabilities, tremors, or those who simply prefer keyboard input to play the entire game.

## User Story

As a student who cannot use a mouse or prefers keyboard input, I want to navigate all screens and interact with all game elements using only the keyboard so that I can play independently.

from typing import List, Optional, Dict
from abc import ABC, abstractmethod

class Focusable(ABC):
    """Mixin/base class for elements that can receive keyboard focus"""
    
    def __init__(self):
        self._is_focused = False
        self._tab_index = 0
        self._can_focus = True
        
    @property
    def is_focused(self) -> bool:
        return self._is_focused and self._can_focus
    
    @property
    def tab_index(self) -> int:
        return self._tab_index
    
    @property
    def can_focus(self) -> bool:
        return self._can_focus
    
    @can_focus.setter
    def can_focus(self, value: bool):
        self._can_focus = value
        if not value and self._is_focused:
            self._is_focused = False
            self._on_focus_lost()
    
    @abstractmethod
    def on_focus(self) -> None:
        """Called when this element receives focus"""
        pass
    
    @abstractmethod
    def on_focus_lost(self) -> None:
        """Called when this element loses focus"""
        pass
    
    @abstractmethod
    def on_activate(self) -> None:
        """Called when activated (e.g., Enter key)"""
        pass
    
    def on_key_down(self, key: int) -> bool:
        """Handle key press while focused. Returns True if handled."""
        return False
    
    def _on_focus(self) -> None:
        self._is_focused = True
        self.on_focus()
        
    def _on_focus_lost(self) -> None:
        self._is_focused = False
        self.on_focus_lost()


class KeyboardNavigator:
    """Manages keyboard focus and navigation"""
    
    def __init__(self):
        self.focusable_elements: List[Focusable] = []
        self.current_focus: Optional[Focusable] = None
        self.navigation_order = []  # Pre-computed focus order
        
    def register(self, element: Focusable) -> None:
        """Register a focusable element"""
        if element not in self.focusable_elements:
            self.focusable_elements.append(element)
            self._rebuild_navigation_order()
            
    def unregister(self, element: Focusable) -> None:
        """Unregister a focusable element"""
        if element in self.focusable_elements:
            self.focusable_elements.remove(element)
            self._rebuild_navigation_order()
            
    def _rebuild_navigation_order(self) -> None:
        """Sort elements by tab_index"""
        self.navigation_order = sorted(
            [e for e in self.focusable_elements if e.can_focus],
            key=lambda e: e.tab_index
        )
        
    def focus_next(self) -> None:
        """Move focus to next element"""
        if not self.navigation_order:
            return
            
        if self.current_focus is None:
            self._focus_element(self.navigation_order[0])
            return
            
        current_index = self.navigation_order.index(self.current_focus)
        next_index = (current_index + 1) % len(self.navigation_order)
        self._focus_element(self.navigation_order[next_index])
        
    def focus_previous(self) -> None:
        """Move focus to previous element"""
        if not self.navigation_order:
            return
            
        if self.current_focus is None:
            self._focus_element(self.navigation_order[-1])
            return
            
        current_index = self.navigation_order.index(self.current_focus)
        prev_index = (current_index - 1) % len(self.navigation_order)
        self._focus_element(self.navigation_order[prev_index])
        
    def _focus_element(self, element: Focusable) -> None:
        """Set focus to element, losing focus from previous"""
        if self.current_focus:
            self.current_focus._on_focus_lost()
            
        self.current_focus = element
        element._on_focus()
        
    def activate_current(self) -> None:
        """Activate the currently focused element"""
        if self.current_focus:
            self.current_focus.on_activate()
            
    def handle_key(self, key: int) -> bool:
        """Handle keyboard input. Returns True if handled."""
        # First, let focused element handle the key
        if self.current_focus:
            if self.current_focus.on_key_down(key):
                return True
                
        # Then handle global navigation
        if key == pygame.K_TAB:
            self.focus_next()
            return True
        elif key == pygame.K_TAB and pygame.key.get_mods() & pygame.KMOD_SHIFT:
            self.focus_previous()
            return True
        elif key == pygame.K_RETURN or key == pygame.K_KP_ENTER:
            self.activate_current()
            return True
        elif key == pygame.K_ESCAPE:
            self._handle_escape()
            return True
            
        return False
        
    def _handle_escape(self) -> None:
        """Handle Escape key (close dialog, cancel action)"""
        if self.current_focus:
            # Let focused element handle first
            pass  # Could add on_escape() callback
            
    def set_focus(self, element: Focusable) -> None:
        """Set focus to specific element"""
        if element in self.focusable_elements and element.can_focus:
            self._focus_element(element)

import pygame

class FocusIndicator:
    """Renders focus indicators around focused elements"""
    
    def __init__(self, screen: pygame.Surface):
        self.screen = screen
        self.indicator_color = (255, 215, 0)  # Gold
        self.indicator_width = 3
        self.border_radius = 8
        self.pulse_alpha = 180
        self.pulse_direction = 1
        self.pulse_speed = 5
        
    def render(self, rect: pygame.Rect) -> None:
        """Draw focus indicator around rect"""
        # Create main border
        border_rect = rect.inflate(self.indicator_width * 2, self.indicator_width * 2)
        
        # Draw outer glow (semi-transparent)
        glow_surface = pygame.Surface((border_rect.width + 8, border_rect.height + 8), pygame.SRCALPHA)
        glow_color = (*self.indicator_color, self.pulse_alpha)
        pygame.draw.rect(glow_surface, glow_color, glow_surface.get_rect(), 
                        width=4, border_radius=self.border_radius + 2)
        glow_rect = glow_surface.get_rect(center=border_rect.center)
        self.screen.blit(glow_surface, glow_rect)
        
        # Draw solid inner border
        pygame.draw.rect(self.screen, self.indicator_color, border_rect,
                        width=self.indicator_width, border_radius=self.border_radius)
        
    def update(self, delta_time: float) -> None:
        """Update animation (pulse effect)"""
        self.pulse_alpha += self.pulse_speed * delta_time * 60
        if self.pulse_alpha >= 220:
            self.pulse_direction = -1
        elif self.pulse_alpha <= 120:
            self.pulse_direction = 1

from typing import Dict, Callable, Optional
import pygame

class Keybindings:
    """Manage keyboard shortcut mappings"""
    
    # Default bindings
    DEFAULT_BINDINGS = {
        'navigate_next': pygame.K_TAB,
        'navigate_previous': (pygame.K_TAB, pygame.KMOD_SHIFT),
        'activate': [pygame.K_RETURN, pygame.K_KP_ENTER],
        'escape': pygame.K_ESCAPE,
        'move_left': pygame.K_LEFT,
        'move_right': pygame.K_RIGHT,
        'move_up': pygame.K_UP,
        'move_down': pygame.K_DOWN,
        'submit_answer': [pygame.K_RETURN, pygame.K_KP_ENTER],
        'delete_char': [pygame.K_BACKSPACE, pygame.K_DELETE],
        'toggle_settings': pygame.K_F1,
        'toggle_captions': pygame.K_F2,
        'toggle_tts': pygame.K_F3,
        'pause_game': pygame.K_p,
        'help': pygame.K_F1,
    }
    
    def __init__(self):
        self.bindings: Dict[str, any] = self.DEFAULT_BINDINGS.copy()
        self.callbacks: Dict[str, Callable] = {}
        
    def bind(self, action: str, key: int, mods: int = 0) -> None:
        """Bind an action to a key"""
        self.bindings[action] = (key, mods) if mods else key
        
    def get_key(self, action: str) -> any:
        """Get key for an action"""
        return self.bindings.get(action)
        
    def register_callback(self, action: str, callback: Callable) -> None:
        """Register callback for an action"""
        self.callbacks[action] = callback
        
    def handle_event(self, event: pygame.Event) -> bool:
        """Handle a key event. Returns True if action matched."""
        if event.type != pygame.KEYDOWN:
            return False
            
        key = event.key
        mods = pygame.key.get_mods()
        
        for action, binding in self.bindings.items():
            if isinstance(binding, tuple):
                expected_key, expected_mods = binding
                if key == expected_key and (mods & expected_mods):
                    self._trigger_action(action)
                    return True
            elif isinstance(binding, list):
                if key in binding:
                    self._trigger_action(action)
                    return True
            else:
                if key == binding:
                    self._trigger_action(action)
                    return True
                    
        return False
        
    def _trigger_action(self, action: str) -> None:
        """Trigger action callback"""
        if action in self.callbacks:
            self.callbacks[action]()

## Notes & Considerations

### Edge Cases

- **Focus Lost On Destroy**: When focused element is destroyed, focus next element
- **Empty Screen**: If no focusable elements, don't crash
- **Modal Dialogs**: Focus should be trapped within modal
- **Dynamic Elements**: New elements should be registered automatically
- **Scrollable Areas**: Focus should scroll into view

### Implementation Tips

1. **Consistent Patterns**: Use same focus order pattern across all screens
2. **Fallback Handling**: If key binding conflicts, prioritize game control
3. **Documentation**: Create keyboard shortcuts reference for users
4. **Testing**: Keep a keyboard available for continuous testing

### Accessibility Best Practices

- **FocusVisible**: Never remove outline without replacement
- **FocusOrder**: Follow visual layout (left-to-right, top-to-bottom)
- **NoTraps**: Users must always be able to navigate away
- **Shortcuts**: Avoid single-letter shortcuts (conflict with typing)
