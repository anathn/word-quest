## Overview

This story implements comprehensive session data collection that tracks all relevant metrics during gameplay for later analysis and progress visualization.

## User Story

As a parent and student, I want detailed data about each practice session so that we can understand performance patterns and improvement over time.

```python
# src/components/session_tracker.py
from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Dict

@dataclass
class WordAttempt:
    word: str
    correct: bool
    first_attempt_correct: bool
    total_attempts: int
    hints_used: int
    time_seconds: float
    timestamp: datetime = field(default_factory=datetime.now)

@dataclass
class SessionSummary:
    session_id: str
    student_id: str
    start_time: datetime
    end_time: datetime = None
    planet_completed: str = None
    words: List[WordAttempt] = field(default_factory=list)
    
    @property
    def duration_seconds(self) -> float:
        if self.end_time:
            return (self.end_time - self.start_time).total_seconds()
        return 0
    
    @property
    def accuracy(self) -> float:
        if not self.words:
            return 0
        return sum(1 for w in self.words if w.correct) / len(self.words)
    
    @property
    def best_streak(self) -> int:
        max_streak = 0
        current_streak = 0
        for word in self.words:
            if word.first_attempt_correct:
                current_streak += 1
                max_streak = max(max_streak, current_streak)
            else:
                current_streak = 0
        return max_streak

class SessionTracker:
    def __init__(self, student_id: str):
        self.current_session = SessionSummary(
            session_id=self.generate_session_id(),
            student_id=student_id,
            start_time=datetime.now()
        )
        
    def record_word_attempt(self, word: str, correct: bool, 
                           attempts: int, hints: int, time_spent: float):
        attempt = WordAttempt(
            word=word,
            correct=correct,
            first_attempt_correct=(attempts == 1),
            total_attempts=attempts,
            hints_used=hints,
            time_seconds=time_spent
        )
        self.current_session.words.append(attempt)
        
    def complete_session(self, planet: str = None):
        self.current_session.end_time = datetime.now()
        self.current_session.planet_completed = planet
        return self.current_session
```
