## Overview

Implement parent-facing weekly progress summary view showing words mastered, practice needed, accuracy trends, and session statistics.

## User Story

As a parent, I want to see a weekly summary of my child's progress so that I can understand their learning and identify areas needing attention.

## Acceptance Criteria

### Functional Requirements

- [ ] Display current week's summary by default
- [ ] Navigate to previous weeks (min of 4 weeks history)
- [ ] Show words mastered count with trend arrow
- [ ] Show accuracy percentage with trend indicator
- [ ] Display list of words mastered this week
- [ ] Display list of words needing practice (low accuracy)
- [ ] Show total practice time for the week
- [ ] Display session count and average session length
- [ ] Show accuracy trend graph (previous 4 weeks)
- [ ] Export summary as PDF or image (optional, nice-to-have)

### Non-Functional Requirements

- [ ] Performance: Summary loads in < 500ms
- [ ] Accessibility: All data screen-reader accessible
- [ ] Error handling: Graceful fallback if data incomplete
- [ ] Visual: Clear, parent-friendly layout with encouraging language

# weekly_summary_generator.py
from datetime import datetime, timedelta, date
from typing import List, Optional
from src.models.summary_data import WeeklySummary, Trend
from src.models.session_data import SessionData  # From EPIC-002

class WeeklySummaryGenerator:
    def __init__(self, session_manager: SessionData):
        self.session_manager = session_manager
    
    def get_week_for_date(self, target_date: date) -> tuple[date, date]:
        """Get start (Monday) and end (Sunday) dates for a week."""
        start = target_date - timedelta(days=target_date.weekday())
        end = start + timedelta(days=6)
        return start, end
    
    def generate_summary(self, student_id: str, target_date: Optional[date] = None) -> WeeklySummary:
        """Generate weekly summary for a student."""
        if target_date is None:
            target_date = date.today()
        
        week_start, week_end = self.get_week_for_date(target_date)
        
        # Get all sessions for this week and student
        sessions = self.session_manager.get_sessions(
            student_id=student_id,
            start_date=week_start,
            end_date=week_end + timedelta(days=1)  # Include entire end day
        )
        
        # Calculate metrics
        words_per_session = []
        correct_answers = 0
        total_attempts = 0
        total_time = 0
        word_accuracy = {}  # word -> {correct: int, total: int}
        best_streak = 0
        current_streak = 0
        
        for session in sessions:
            total_time += session.duration_seconds / 60
            
            for word_result in session.word_results:
                words_per_session.append(word_result.word)
                total_attempts += word_result.attempts
                if word_result.correct_on_first_attempt:
                    correct_answers += 1
                    current_streak += 1
                    best_streak = max(best_streak, current_streak)
                else:
                    current_streak = 0
                
                # Track per-word accuracy
                word = word_result.word
                if word not in word_accuracy:
                    word_accuracy[word] = {"correct": 0, "total": 0}
                word_accuracy[word]["total"] += 1
                if word_result.attempts == 1:
                    word_accuracy[word]["correct"] += 1
        
        # Calculate derived metrics
        words_practiced = len(set(words_per_session)) if words_per_session else 0
        accuracy_rate = correct_answers / total_attempts if total_attempts > 0 else 0
        
        # Words mastered (80%+ accuracy, at least 2 attempts)
        mastered = [
            word for word, data in word_accuracy.items()
            if data["total"] >= 2 and (data["correct"] / data["total"]) >= 0.8
        ]
        
        # Words needing practice (<50% accuracy, at least 3 attempts)
        needing_practice = [
            word for word, data in word_accuracy.items()
            if data["total"] >= 3 and (data["correct"] / data["total"]) < 0.5
        ]
        
        summary = WeeklySummary(
            week_start=week_start,
            week_end=week_end,
            student_id=student_id,
            words_mastered=len(mastered),
            words_practiced=words_practiced,
            accuracy_rate=accuracy_rate,
            total_sessions=len(sessions),
            total_time_minutes=int(total_time),
            best_streak=best_streak,
            avg_session_length=total_time / len(sessions) if sessions else 0,
            words_mastered_list=sorted([w.upper() for w in mastered])[:10],  # Top 10
            words_needing_practice=sorted([w.upper() for w in needing_practice])[:10]
        )
        
        # Add trends
        summary.trend_accuracy = self._calculate_accuracy_trend(student_id, week_start)
        summary.trend_mastered = self._calculate_mastered_trend(student_id, week_start)
        
        return summary
    
    def _calculate_accuracy_trend(self, student_id: str, current_week_start: date) -> Optional[Trend]:
        """Compare accuracy to previous week."""
        prev_week_start = current_week_start - timedelta(weeks=1)
        prev_summary = self.generate_summary(student_id, prev_week_start)
        current_summary = self.generate_summary(student_id, current_week_start)
        
        if prev_summary.accuracy_rate == 0 and current_summary.accuracy_rate == 0:
            return Trend.STABLE
        
        diff = current_summary.accuracy_rate - prev_summary.accuracy_rate
        if diff > 0.05:
            return Trend.IMPROVING
        elif diff < -0.05:
            return Trend.DECLINING
        else:
            return Trend.STABLE
    
    def _calculate_mastered_trend(self, student_id: str, current_week_start: date) -> Optional[Trend]:
        """Compare words mastered to previous week."""
        prev_summary = self.generate_summary(student_id, current_week_start - timedelta(weeks=1))
        current_summary = self.generate_summary(student_id, current_week_start)
        
        diff = current_summary.words_mastered - prev_summary.words_mastered
        if diff > 2:
            return Trend.IMPROVING
        elif diff < -2:
            return Trend.DECLINING
        else:
            return Trend.STABLE
    
    def get_weekly_history(self, student_id: str, weeks: int = 4) -> List[WeeklySummary]:
        """Get summary for previous N weeks including current."""
        history = []
        for i in range(weeks):
            target_date = date.today() - timedelta(weeks=i)
            summary = self.generate_summary(student_id, target_date)
            history.append(summary)
        return sorted(history, key=lambda s: s.week_start)
