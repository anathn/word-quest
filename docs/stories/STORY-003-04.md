## Overview

Implement ability to import spelling word lists from CSV files, supporting common school formats with automatic difficulty assignment.

## User Story

As a parent, I want to import spelling lists from CSV files so that I can quickly add my child's school vocabulary without manual entry.

## Acceptance Criteria

### Functional Requirements

- [ ] Import CSV file via file picker dialog
- [ ] Support CSV with headers (word, definition, difficulty columns)
- [ ] Support CSV with minimal format (word column only)
- [ ] Auto-detect CSV delimiter (comma, semicolon, tab)
- [ ] Map CSV columns to word fields
- [ ] Auto-assign difficulty based on word length or column value
- [ ] Preview imported words before confirming
- [ ] Show import summary (X words added, Y skipped)
- [ ] Handle duplicate words (skip or overwrite option)
- [ ] Provide sample CSV template download
- [ ] Validation error feedback for malformed rows

### Non-Functional Requirements

- [ ] Performance: Parse 1000+ word files in < 1 second
- [ ] Accessibility: File picker keyboard accessible
- [ ] Error handling: Graceful handling of encoding issues, malformed files
- [ ] Storage: Validated data stored in standard format

# csv_importer.py
import csv
import chardet  # For encoding detection
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass
from enum import Enum

class Difficulty(Enum):
    BEGINNER = "beginner"
    MEDIUM = "medium"
    ADVANCED = "advanced"

@dataclass
class ParsedWord:
    spelling: str
    definition: str
    difficulty: Difficulty
    row_number: int
    error: Optional[str] = None

@dataclass
class ImportResult:
    successful: List[ParsedWord]
    skipped: List[Tuple[int, str]]  # (row_number, reason)
    total_processed: int
    
    @property
    def success_count(self) -> int:
        return len(self.successful)
    
    @property
    def skip_count(self) -> int:
        return len(self.skipped)

class CSVImporter:
    # Column name variations to recognize
    WORD_COLUMNS = ["word", "spelling", "term", "vocabulary"]
    DEF_COLUMNS = ["definition", "definition", "meaning", "context"]
    DIFF_COLUMNS = ["difficulty", "level", "grade", "category"]
    
    def __init__(self, filepath: str):
        self.filepath = filepath
        self.encodings = ["utf-8", "latin-1", "cp1252", "iso-8859-1"]
    
    def detect_encoding(self) -> str:
        """Auto-detect file encoding."""
        with open(self.filepath, "rb") as f:
            result = chardet.detect(f.read(10000))
            return result["encoding"] or "utf-8"
    
    def detect_delimiter(self, sample: str) -> str:
        """Auto-detect CSV delimiter."""
        delimiters = [",", ";", "\t"]
        counts = {d: sample.count(d) for d in delimiters}
        return max(counts, key=counts.get)
    
    def parse_header(self, row: List[str]) -> Dict[str, int]:
        """Identify column indices for word, definition, difficulty."""
        mapping = {}
        for i, cell in enumerate(row):
            header = cell.lower().strip()
            if header in self.WORD_COLUMNS:
                mapping["word"] = i
            elif header in self.DEF_COLUMNS:
                mapping["definition"] = i
            elif header in self.DIFF_COLUMNS:
                mapping["difficulty"] = i
        return mapping
    
    def _auto_assign_difficulty(self, spelling: str) -> Difficulty:
        """Assign difficulty based on word length."""
        length = len(spelling)
        if length <= 5:
            return Difficulty.BEGINNER
        elif length <= 8:
            return Difficulty.MEDIUM
        else:
            return Difficulty.ADVANCED
    
    def _normalize_difficulty(self, value: str) -> Optional[Difficulty]:
        """Parse difficulty value from CSV."""
        if not value:
            return None
        value = value.lower().strip()
        mapping = {
            "beginner": Difficulty.BEGINNER,
            "easy": Difficulty.BEGINNER,
            "1": Difficulty.BEGINNER,
            "medium": Difficulty.MEDIUM,
            "moderate": Difficulty.MEDIUM,
            "2": Difficulty.MEDIUM,
            "advanced": Difficulty.ADVANCED,
            "difficult": Difficulty.ADVANCED,
            "hard": Difficulty.ADVANCED,
            "3": Difficulty.ADVANCED,
        }
        return mapping.get(value)
    
    def parse_file(self) -> ImportResult:
        """Parse CSV file and return import results."""
        successful = []
        skipped = []
        
        try:
            encoding = self.detect_encoding()
        except Exception as e:
            return ImportResult([], [(0, f"Could not detect encoding: {e}")], 0)
        
        try:
            with open(self.filepath, "r", encoding=encoding) as f:
                content = f.read()
        except Exception as e:
            return ImportResult([], [(0, f"Could not read file: {e}")], 0)
        
        lines = content.strip().split("\n")
        if not lines:
            return ImportResult([], [(0, "File is empty")], 0)
        
        # Detect delimiter from first line
        delimiter = self.detect_delimiter(lines[0])
        
        try:
            reader = csv.reader(lines, delimiter=delimiter)
            rows = list(reader)
        except Exception as e:
            return ImportResult([], [(0, f"CSV parsing error: {e}")], 0)]
        
        if not rows:
            return ImportResult([], [(0, "No data found")], 0)
        
        # Check if first row is header
        header = None
        start_row = 0
        first_row_lower = [c.lower().strip() for c in rows[0]]
        if any(any(header_var in cell for header_var in self.WORD_COLUMNS) for cell in first_row_lower):
            header = self.parse_header(rows[0])
            start_row = 1
        else:
            # No header, assume first column is word
            header = {"word": 0}
            start_row = 0
        
        # Parse data rows
        for i, row in enumerate(rows[start_row:], start=start_row + 1):
            if not row or all(not cell.strip() for cell in row):
                continue  # Skip empty rows
            
            parsed = self._parse_row(row, header, i)
            if parsed.error:
                skipped.append((parsed.row_number, parsed.error))
            else:
                successful.append(parsed)
        
        return ImportResult(successful, skipped, i)
    
    def _parse_row(self, row: List[str], header: Dict[str, int], row_num: int) -> ParsedWord:
        """Parse a single row."""
        try:
            # Get word
            word_idx = header.get("word", 0)
            if word_idx >= len(row):
                return ParsedWord("", "", Difficulty.MEDIUM, row_num, "Missing word")
            
            spelling = row[word_idx].strip().upper()
            if not spelling or not spelling.replace("-", "").isalpha():
                return ParsedWord("", "", Difficulty.MEDIUM, row_num, "Invalid word format")
            
            if len(spelling) < 2 or len(spelling) > 25:
                return ParsedWord("", "", Difficulty.MEDIUM, row_num, "Word too long or short")
            
            # Get definition (optional)
            def_idx = header.get("definition")
            definition = row[def_idx].strip() if def_idx is not None and def_idx < len(row) else ""
            
            # Get difficulty
            diff_idx = header.get("difficulty")
            difficulty = None
            if diff_idx is not None and diff_idx < len(row):
                difficulty = self._normalize_difficulty(row[diff_idx])
            if difficulty is None:
                difficulty = self._auto_assign_difficulty(spelling)
            
            return ParsedWord(spelling, definition, difficulty, row_num)
        
        except Exception as e:
            return ParsedWord("", "", Difficulty.MEDIUM, row_num, f"Parse error: {e}")
