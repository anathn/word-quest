## Overview

Implement the OpenDyslexic font as an optional font choice for the game, providing better readability for students with dyslexia. The font features weighted bottoms, unique letter shapes, and increased letter spacing to reduce letter confusion and improve reading fluency.

## User Story

As a student with dyslexia, I want to use a font designed for dyslexic readers so that letters don't swap around or look confusing when I'm spelling.

## Acceptance Criteria

# src/ui/font_manager.py
import pygame
from typing import Dict, Optional
from pathlib import Path

class FontManager:
    """Manages game fonts with support for multiple font families"""
    
    # Font family names
    FONT_DEFAULT = "default"
    FONT_ODL = "opendyslexic"  # OpenDyslexic
    
    # Font file paths
    DEFAULT_FONT_PATH = "assets/fonts/Roboto-Regular.ttf"
    ODL_REGULAR_PATH = "assets/fonts/OpenDyslexic-Regular.ttf"
    ODL_BOLD_PATH = "assets/fonts/OpenDyslexic-Bold.ttf"
    
    def __init__(self):
        self.font_families: Dict[str, Dict[int, pygame.font.Font]] = {}
        self.current_family = self.FONT_DEFAULT
        self._ensure_default_font()
        
    def _ensure_default_font(self) -> None:
        """Ensure default font is always available"""
        try:
            # Try loading custom default font
            self.font_families[self.FONT_DEFAULT] = {}
            if Path(self.DEFAULT_FONT_PATH).exists():
                self._load_font(self.FONT_DEFAULT, 24, self.DEFAULT_FONT_PATH)
            else:
                # Use pygame default as fallback
                self.font_families[self.FONT_DEFAULT][24] = pygame.font.Font(None, 24)
        except Exception as e:
            print(f"Warning: Could not load default font: {e}")
            # Ultimate fallback
            self.font_families[self.FONT_DEFAULT][24] = pygame.font.Font(None, 24)
            
    def _load_font(self, family: str, size: int, path: str) -> pygame.font.Font:
        """Load a font file at specified size"""
        if not Path(path).exists():
            raise FileNotFoundError(f"Font file not found: {path}")
            
        return pygame.font.Font(path, size)
        
    def initialize_opendyslexic(self) -> bool:
        """Initialize OpenDyslexic font family"""
        try:
            if self.FONT_ODL not in self.font_families:
                self.font_families[self.FONT_ODL] = {}
                
            # Load both regular and bold
            if Path(self.ODL_REGULAR_PATH).exists():
                # Cache fonts at common sizes (pygame caches internally)
                for size in [18, 24, 36, 48, 64]:
                    self.font_families[self.ODL][size] = self._load_font(
                        self.ODL, size, self.ODL_REGULAR_PATH
                    )
                return True
            else:
                print(f"OpenDyslexic font not found at {self.ODL_REGULAR_PATH}")
                return False
        except Exception as e:
            print(f"Failed to load OpenDyslexic font: {e}")
            return False
            
    def get_font(self, family: str = None, size: int = 24, 
                bold: bool = False, italic: bool = False) -> pygame.font.Font:
        """Get font for specified family and size"""
        family = family or self.current_family
        
        if family not in self.font_families:
            # Fall back to default
            family = self.FONT_DEFAULT
            
        # Find closest cached size or load new
        if size not in self.font_families[family]:
            self._load_font_to_cache(family, size, bold)
            
        return self.font_families[family].get(size, self._get_fallback_font(size))
        
    def _load_font_to_cache(self, family: str, size: int, bold: bool = False) -> None:
        """Load font at specified size and cache it"""
        if family == self.FONT_ODL:
            path = self.ODL_BOLD_PATH if bold else self.ODL_REGULAR_PATH
        else:
            path = self.DEFAULT_FONT_PATH
            
        try:
            font = self._load_font(family, size, path)
            # pygame doesn't support modifying loaded fonts, so we just cache
            self.font_families[family][size] = font
        except:
            # Use pygame default as ultimate fallback
            self.font_families[family][size] = pygame.font.Font(None, size)
            
    def _get_fallback_font(self, size: int) -> pygame.font.Font:
        """Get fallback font (pygame default)"""
        return pygame.font.Font(None, size)
        
    def set_current_family(self, family: str) -> bool:
        """Set current font family. Returns True if successful."""
        if family in self.font_families:
            self.current_family = family
            return True
        return False
        
    def is_opendyslexic_available(self) -> bool:
        """Check if OpenDyslexic font is loaded"""
        return (self.FONT_ODL in self.font_families and 
                len(self.font_families[self.FONT_ODL]) > 0)

from dataclasses import dataclass
from typing import Optional

@dataclass
class TypographySettings:
    """Typography configuration"""
    font_family: str = "default"  # "default" or "opendyslexic"
    font_size_base: int = 24
    font_size_large: int = 48
    font_size_small: int = 18
    letter_spacing: int = 2
    word_spacing: int = 4
    line_height: float = 1.5
    
    def to_dict(self) -> dict:
        return {
            'font_family': self.font_family,
            'font_size_base': self.font_size_base,
            'font_size_large': self.font_size_large,
            'font_size_small': self.font_size_small,
            'letter_spacing': self.letter_spacing,
            'word_spacing': self.word_spacing,
            'line_height': self.line_height
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> 'TypographySettings':
        return cls(
            font_family=data.get('font_family', 'default'),
            font_size_base=data.get('font_size_base', 24),
            font_size_large=data.get('font_size_large', 48),
            font_size_small=data.get('font_size_small', 18),
            letter_spacing=data.get('letter_spacing', 2),
            word_spacing=data.get('word_spacing', 4),
            line_height=data.get('line_height', 1.5)
        )

class ParentDashboard:
    """Parent dashboard screen with font settings"""
    
    def __init__(self, screen: pygame.Surface, font_manager: FontManager):
        self.screen = screen
        self.font_manager = font_manager
        self.typography_settings = self._load_typography_settings()
        self._setup_ui()
        
    def _setup_ui(self):
        """Setup UI with current font"""
        # Apply font setting
        self.font_manager.set_current_family(self.typography_settings.font_family)
        
        # Create font-based UI elements
        self.header_font = self.font_manager.get_font(size=48, bold=True)
        self.body_font = self.font_manager.get_font(size=24)
        self.small_font = self.font_manager.get_font(size=18)
        
    def _render_font_settings_panel(self):
        """Render font selection panel"""
        # Font selection dropdown
        font_options = [
            ("Default Font", "default"),
            ("Dyslexia-Friendly (OpenDyslexic)", "opendyslexic")
        ]
        
        # Sample text showing both fonts
        sample_text = "The quick brown fox jumps over the lazy dog"
        
        # Compare display
        self._draw_sample_comparison(sample_text)
        
    def _draw_sample_comparison(self, text: str):
        """Draw side-by-side font comparison"""
        # Default font sample
        default_surf = self.font_manager.get_font("default", 24).render(text, True, (255, 255, 255))
        self.screen.blit(default_surf, (50, 200))
        
        # OpenDyslexic sample (if available)
        if self.font_manager.is_opendyslexic_available():
            odl_surf = self.font_manager.get_font("opendyslexic", 24).render(text, True, (255, 255, 255))
            self.screen.blit(odl_surf, (50, 250))

## OpenDyslexic Font Information

**What is OpenDyslexic?**
OpenDyslexic is a free, open-source typeface designed to increase readability for people with dyslexia. It features:
- Weighted bottoms to guide the eye
- Unique letter shapes to prevent swapping (b/d, p/q)
- Increased letter spacing
- Heavier starting letters for each word

**License:** OpenDyslexic is released under the MIT License - free for personal and commercial use.

**Download:** https://github.com/abarrq/OpenDyslexic

**Research:** Studies show dyslexic fonts can improve reading speed and comprehension for users with dyslexia, though effects vary by individual.

### When to Use

- **Recommended**: For students with diagnosed dyslexia or reading difficulties
- **Optional**: Any student may prefer this font style
- **Not Required**: Students without dyslexia may find standard fonts easier to read

### Letter Distinguishment

OpenDyslexic helps distinguish commonly confused letters:

| Letter Pair | Standard Font | OpenDyslexic |
|-------------|---------------|--------------|
| b / d | Similar | Different bottom weights |
| p / q | Similar | Different bottom weights |
| m / n | Similar | Different letter shapes |
| u / n | Similar | Different letter shapes |

### Accessibility Guideline

This font option supports WCAG 2.1 Success Criterion 1.4.4 (Resize Text) and 1.4.12 (Text Spacing).
