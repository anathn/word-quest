## Overview

Integrate a text-to-speech (TTS) engine to provide audio pronunciation for all spelling words, with adjustable speech speed and volume controls. This enables students with reading difficulties, visual impairments, or auditory learning preferences to access all game content.

## User Story

As a student with dyslexia or visual impairment, I want to hear words spoken aloud so that I can focus on spelling without struggling to read.

# src/components/tts_engine.py
import pyttsx3
from abc import ABC, abstractmethod
from typing import Optional
from threading import Lock

class TTSEngine(ABC):
    """Abstract base class for text-to-speech engines"""
    
    @abstractmethod
    def initialize(self) -> bool:
        """Initialize the TTS engine. Returns True on success."""
        pass
    
    @abstractmethod
    def speak(self, text: str, speed: float = 1.0, 
              volume: float = 1.0) -> None:
        """Speak text at given speed and volume"""
        pass
    
    @abstractmethod
    def stop(self) -> None:
        """Stop current speech"""
        pass
    
    @abstractmethod
    def is_speaking(self) -> bool:
        """Check if currently speaking"""
        pass
    
    @abstractmethod
    def get_available_voices(self) -> list:
        """Get list of available voices"""
        pass

class PlatformTTSWrapper:
    """Platform-specific TTS implementation using pyttsx3"""
    
    def __init__(self):
        self.engine = None
        self.initialized = False
        self._lock = Lock()
        
    def initialize(self) -> bool:
        """Initialize pyttsx3 engine"""
        try:
            with self._lock:
                if not self.initialized:
                    self.engine = pyttsx3.init()
                    self.engine.setProperty('rate', 200)  # words per minute
                    self.engine.setProperty('volume', 1.0)
                    self.initialized = True
            return True
        except Exception as e:
            print(f"TTS initialization failed: {e}")
            return False
    
    def speak(self, text: str, speed: float = 1.0, volume: float = 1.0) -> None:
        """Speak text with adjustable speed and volume"""
        if not self.initialized:
            return
            
        with self._lock:
            try:
                # Adjust properties
                base_rate = self.engine.getProperty('rate')
                self.engine.setProperty('rate', int(base_rate * speed))
                self.engine.setProperty('volume', volume)
                
                # Speak
                self.engine.say(text)
                self.engine.runAndWait()
            except Exception as e:
                print(f"TTS speak error: {e}")
    
    def stop(self) -> None:
        """Stop current speech"""
        if self.initialized and self.is_speaking():
            with self._lock:
                self.engine.stop()
    
    def is_speaking(self) -> bool:
        """Check if currently speaking"""
        if not self.initialized:
            return False
        # Check if engine is processing
        try:
            return self.engine.is_playing()
        except:
            return False
    
    def get_available_voices(self) -> list:
        """Get list of available voices"""
        if not self.initialized:
            return []
        return self.engine.getProperty('voices')


# src/components/tts_manager.py
from typing import Queue
from threading import Thread

class TTSManager:
    """Manages TTS queue and game integration"""
    
    def __init__(self, engine: TTSEngine):
        self.engine = engine
        self.speech_queue = Queue()
        self.enabled = True
        self.speed = 1.0
        self.volume = 1.0
        self._speaker_thread = None
        self._running = False
        
    def start(self) -> None:
        """Start TTS speaker thread"""
        self._running = True
        self._speaker_thread = Thread(target=self._speaker_loop, daemon=True)
        self._speaker_thread.start()
        
    def stop(self) -> None:
        """Stop TTS and clear queue"""
        self._running = False
        self.speech_queue.clear()
        self.engine.stop()
        
    def speak_word(self, word: str, spell_out: bool = False) -> None:
        """Speak a word, optionally spelling it letter by letter"""
        if not self.enabled:
            return
            
        if spell_out:
            text = " ".join(word.upper())
        else:
            text = word
            
        self.speech_queue.put(text)
        
    def speak_letters(self, word: str) -> None:
        """Spell word letter by letter"""
        if not self.enabled:
            return
            
        letters = " ".join(word.upper())
        self.speech_queue.put(letters)
        
    def _speaker_loop(self) -> None:
        """Background thread that processes speech queue"""
        while self._running:
            try:
                text = self.speech_queue.get(timeout=0.1)
                self.engine.speak(text, self.speed, self.volume)
            except:
                continue

# src/components/tts_settings.py
from dataclasses import dataclass
from typing import Optional
import json

@dataclass
class TTSSettings:
    """TTS configuration"""
    enabled: bool = True
    speed: float = 1.0  # 0.5 to 2.0
    volume: float = 1.0  # 0.0 to 1.0
    voice_id: Optional[str] = None
    
    def to_dict(self) -> dict:
        return {
            'enabled': self.enabled,
            'speed': self.speed,
            'volume': self.volume,
            'voice_id': self.voice_id
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> 'TTSSettings':
        return cls(
            enabled=data.get('enabled', True),
            speed=data.get('speed', 1.0),
            volume=data.get('volume', 1.0),
            voice_id=data.get('voice_id')
        )
    
    def validate(self) -> bool:
        """Validate settings are in acceptable ranges"""
        return (0.5 <= self.speed <= 2.0 and 
                0.0 <= self.volume <= 1.0)

class TTSConfigManager:
    """Persistence layer for TTS settings"""
    
    def __init__(self, config_path: str = "data/tts_config.json"):
        self.config_path = config_path
        self.settings = TTSSettings()
        self.load()
        
    def load(self) -> None:
        """Load settings from file"""
        try:
            with open(self.config_path, 'r') as f:
                data = json.load(f)
                self.settings = TTSSettings.from_dict(data.get('tts', {}))
        except FileNotFoundError:
            pass  # Use defaults
        except json.JSONDecodeError:
            pass  # Use defaults
            
    def save(self) -> None:
        """Save settings to file"""
        import os
        os.makedirs(os.path.dirname(self.config_path), exist_ok=True)
        
        try:
            with open(self.config_path, 'r') as f:
                data = json.load(f)
        except (FileNotFoundError, json.JSONDecodeError):
            data = {}
            
        data['tts'] = self.settings.to_dict()
        
        with open(self.config_path, 'w') as f:
            json.dump(data, f, indent=2)

## Notes & Considerations

### Edge Cases

- **No TTS Available**: Handle gracefully on systems without TTS (show message, disable feature)
- **Long Words**: Words with 15+ letters may take long to spell - consider abbreviation
- **Special Characters**: Word lists may contain apostrophes, hyphens - test pronunciation
- **Numbers**: If word is "8th" - should it say "eighth" or "8-t-h"?
- **Overlapping Speech**: Quick successional words may queue up - implement debounce

### Implementation Tips

1. **Use pyttsx3**: Works offline, cross-platform, no API keys needed
2. **Initialize Early**: Initialize TTS during game startup, not when first needed
3. **Queue System**: Prevent overlapping speech with a queue
4. **Cache Voices**: Load available voices once at startup
5. **Graceful Degradation**: If TTS fails, game still playable (just no audio)

### Performance Considerations

- **Initialization**: TTS can take 1-2 seconds to initialize - do this in background
- **Speech Latency**: First word may have 200-500ms delay - acceptable for MVP
- **Memory**: pyttsx3 uses ~50-100MB RAM
- **CPU**: Minimal impact (<5% during speech)

### Accessibility Best Practices

- **Default Speed**: 1.0x (normal) is best starting point
- **Clear Speech**: TTS may not handle all words perfectly - consider word list review
- **Visual Feedback**: Show TTS is active (speaker icon, animation)
- **No Auto-Restart**: If TTS stops mid-word, don't auto-restart (annoying)

### Privacy Considerations

- **Offline Operation**: All TTS happens locally, no data sent to external services
- **No Voice Recording**: Game does not record student voice (only text-to-speech, not speech-to-text)
- **GDPR Compliant**: No PII sent to TTS service
