## Overview

This story allows students to personalize their gaming experience by choosing their rocket ship's color. This simple customization feature creates a sense of ownership and makes the game more engaging.

## User Story

As a 3rd grade student, I want to choose my rocket's color so that my spaceship feels special and uniquely mine.

# src/ui/color_picker.py
import pygame

class ColorPicker:
    def __init__(self, screen: pygame.Surface, x: int, y: int):
        self.screen = screen
        self.position = (x, y)
        self.swatch_size = (60, 60)
        self.spacing = 10
        self.selected_color = DEFAULT_ROCKET_COLOR
        self.colors = ROCKET_COLOR_PRESETS
        self.on_color_selected: Optional[Callable[[str], None]] = None
        
    def render(self):
        """Draw color swatches in grid"""
        for i, color_data in enumerate(self.colors):
            row = i // 4
            col = i % 4
            x = self.position[0] + col * (self.swatch_size[0] + self.spacing)
            y = self.position[1] + row * (self.swatch_size[1] + self.spacing)
            
            # Draw swatch
            pygame.draw.rect(self.screen, color_data["rgb"], 
                           (x, y, *self.swatch_size))
            
            # Draw border if selected
            if color_data["hex"] == self.selected_color:
                pygame.draw.rect(self.screen, (255, 255, 255),
                               (x - 2, y - 2, *self.swatch_size), 3)
            
            # Draw color name below swatch (optional)
            # font.render(color_data["name"], ...)
    
    def handle_event(self, event: pygame.Event) -> bool:
        """Handle mouse clicks on color swatches"""
        if event.type == pygame.MOUSEBUTTONDOWN:
            for i, color_data in enumerate(self.colors):
                row = i // 4
                col = i % 4
                x = self.position[0] + col * (self.swatch_size[0] + self.spacing)
                y = self.position[1] + row * (self.swatch_size[1] + self.spacing)
                
                # Check if click is within swatch bounds
                rect = pygame.Rect(x, y, *self.swatch_size)
                if rect.collidepoint(event.pos):
                    self.selected_color = color_data["hex"]
                    if self.on_color_selected:
                        self.on_color_selected(color_data["hex"])
                    return True
        return False


# src/components/rocket_renderer.py
class RocketRenderer:
    def __init__(self, screen: pygame.Surface):
        self.screen = screen
        self.base_sprite = pygame.image.load("assets/rocket/base.png")
        self.current_color = DEFAULT_ROCKET_COLOR
        self.tinted_sprite = None
        self._recache_sprite()
        
    def set_color(self, hex_color: str):
        """Change rocket color (tint the sprite)"""
        self.current_color = hex_color
        self._recache_sprite()
        
    def _recache_sprite(self):
        """Apply color tint to base sprite"""
        # Parse hex color to RGB
        rgb = self._hex_to_rgb(self.current_color)
        
        # Load base sprite and apply color tint
        # Simple approach: multiply pixel colors by tint
        self.tinted_sprite = self._apply_tint(self.base_sprite, rgb)
        
    def _apply_tint(self, sprite: pygame.Surface, rgb: tuple) -> pygame.Surface:
        """Apply color tint to sprite"""
        # Create a copy of the sprite
        tinted = sprite.copy()
        
        # Apply color multiplication
        for x in range(tinted.get_width()):
            for y in range(tinted.get_height()):
                pixel = tinted.get_at((x, y))
                # Only tint non-transparent pixels
                if pixel.a > 0:
                    # Multiply RGB by tint (normalize to 0-1, multiply, scale back)
                    r = int(pixel.r * rgb[0] / 255)
                    g = int(pixel.g * rgb[1] / 255)
                    b = int(pixel.b * rgb[2] / 255)
                    tinted.set_at((x, y), pygame.Color(r, g, b, pixel.a))
        
        return tinted
        
    def render(self, position: tuple):
        """Draw tinted rocket at position"""
        self.screen.blit(self.tinted_sprite, position)

class StudentProfile:

        self.name = name
        self.avatar = "default.png"

        profile = cls(data["name"])
        profile.avatar = data.get("avatar", "default.png")
        profile.rocket_color = data.get("rocket_color", DEFAULT_ROCKET_COLOR)
        ...

## Notes & Considerations

- **Performance**: Caching tinted sprite avoids recomputing each frame
- **Color Blindness**: Text labels ensure colors are identifiable
- **Future Expansion**: Could add custom color picker (RGB sliders)
- **Sprite Design**: Base rocket should be monochrome for easy tinting

### Rocket Sprite Design for Tinting

For effective color tinting, the base rocket sprite should:
- Be designed in grayscale or single base color
- Have clear light/dark areas preserved
- Avoid intricate details that might get lost in tinting
- Use alpha transparency for non-rectangular shapes
