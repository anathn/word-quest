```python
"""Entry point for launching Word Quest"""

import pygame
import sys
from src.game import Game
from src.config import WINDOW_TITLE, FPS

def initialize_pygame():
    """Initialize pygame and subsystems"""
    pygame.init()
    pygame.mixer.init()
    return True

def create_display():
    """Create the game window"""
    from src.config import WINDOW_WIDTH, WINDOW_HEIGHT, DEFAULT_WINDOWED
    
    # Flag for windowed vs fullscreen
    flags = pygame.RESIZABLE if DEFAULT_WINDOWED else 0
    
    screen = pygame.display.set_mode(
        (WINDOW_WIDTH, WINDOW_HEIGHT),
        flags
    )
    pygame.display.set_caption(WINDOW_TITLE)
    
    return screen

def main():
    """Main entry point"""
    try:
        # Initialize pygame
        if not initialize_pygame():
            print("Error: Failed to initialize Pygame")
            sys.exit(1)
        
        # Create display
        screen = create_display()
        
        # Create and run game
        game = Game(screen)
        game.run()
        
    except KeyboardInterrupt:
        print("\nGame exiting...")
    except Exception as e:
        print(f"Fatal error: {e}")
        import traceback
        traceback.print_exc()
        sys.exit(1)
    finally:
        pygame.quit()
        sys.exit(0)

if __name__ == "__main__":
    main()
```

```python
# src/game.py
"""Main game class with game loop"""

import pygame
from typing import Optional
from src.screens.main_menu import MainMenu
from src.ui.screen_manager import ScreenManager
from src.components.audio_manager import AudioManager
from src.components.storage_manager import StorageManager

class Game:
    """Main game controller and loop"""
    
    def __init__(self, screen: pygame.Surface):
        self.screen = screen
        self.clock = pygame.time.Clock()
        self.running = True
        self.paused = False
        
        # Initialize managers
        self.screen_manager = ScreenManager(screen)
        self.audio_manager = AudioManager()
        self.storage = StorageManager()
        
        # Initialize initial screen
        self._setup_initial_screen()
        
    def _setup_initial_screen(self):
        """Setup the initial screen (main menu)"""
        # Check if this is first run
        is_first_run = not self.storage.has_profiles()
        
        if is_first_run:
            # Show setup wizard or main menu
            initial_screen = MainMenu(self.screen)
        else:
            initial_screen = MainMenu(self.screen)
            
        self.screen_manager.push_screen(initial_screen)
        
    def run(self):
        """Main game loop"""
        while self.running:
            # Handle events
            for event in pygame.event.get():
                if event.type == pygame.QUIT:
                    self.running = False
                    
                elif event.type == pygame.KEYDOWN:
                    if event.key == pygame.K_ESCAPE:
                        # Handle escape (back to menu, quit, etc.)
                        self._handle_escape()
                        
                    elif event.key == pygame.K_p:
                        # Toggle pause
                        self.paused = not self.paused
                        
                # Pass events to current screen
                current_screen = self.screen_manager.current_screen
                if current_screen:
                    current_screen.handle_event(event)
                    
            if not self.paused:
                # Update current screen
                current_screen = self.screen_manager.current_screen
                if current_screen:
                    current_screen.update()
                    
                # Clear screen
                self.screen.fill((26, 26, 62))  # Background color
                
                # Draw current screen
                current_screen = self.screen_manager.current_screen
                if current_screen:
                    current_screen.draw(self.screen)
                    
                # Update display
                pygame.display.flip()
                
            # Cap frame rate
            self.clock.tick(60)
            
    def _handle_escape(self):
        """Handle Escape key press"""
        current_screen = self.screen_manager.current_screen
        
        if current_screen and hasattr(current_screen, 'on_escape'):
            current_screen.on_escape()
        else:
            # Default: go back one screen or quit
            if len(self.screen_manager.screen_stack) > 1:
                self.screen_manager.pop_screen()
            else:
                self.running = False
                
    def quit(self):
        """Clean up and quit game"""
        self.audio_manager.stop_music()
        self.storage.save_all()
        self.running = False
```

```python
# src/ui/screen_manager.py
"""Manages screen stack and transitions"""

import pygame
from typing import List, Optional
from abc import ABC, abstractmethod

class Screen(ABC):
    """Base class for all game screens"""
    
    def __init__(self, screen: pygame.Surface):
        self.screen = screen
        self.next_screen: Optional[str] = None
        self.prev_screen: Optional[str] = None
        
    @abstractmethod
    def handle_event(self, event: pygame.Event):
        """Handle input events"""
        pass
        
    @abstractmethod
    def update(self):
        """Update game logic"""
        pass
        
    @abstractmethod
    def draw(self, surface: pygame.Surface):
        """Draw screen elements"""
        pass
        
    def on_enter(self):
        """Called when screen becomes active"""
        pass
        
    def on_exit(self):
        """Called when screen is left"""
        pass
        
    def on_escape(self):
        """Handle Escape key (default: go back)"""
        self.next_screen = None  # Signal to pop


class ScreenManager:
    """Manages screen stack"""
    
    def __init__(self, screen: pygame.Surface):
        self.screen = screen  # Main pygame display surface
        self.screen_stack: List[Screen] = []
        self._current_screen: Optional[Screen] = None
        
    @property
    def current_screen(self) -> Optional[Screen]:
        """Get current active screen"""
        return self._current_screen
        
    def push_screen(self, screen: Screen):
        """Push new screen onto stack"""
        if self._current_screen:
            self._current_screen.on_exit()
            
        self.screen_stack.append(screen)
        self._current_screen = screen
        screen.on_enter()
        
    def pop_screen(self) -> Optional[Screen]:
        """Pop current screen, return to previous"""
        if not self.screen_stack:
            return None
            
        if self._current_screen:
            self._current_screen.on_exit()
            
        self.screen_stack.pop()
        
        if self.screen_stack:
            self._current_screen = self.screen_stack[-1]
            self._current_screen.on_enter()
        else:
            self._current_screen = None
            
        return self._current_screen
        
    def replace_screen(self, screen: Screen):
        """Replace current screen without going back"""
        if self._current_screen:
            self._current_screen.on_exit()
            
        self.screen_stack[-1] = screen
        self._current_screen = screen
        screen.on_enter()
```

```python
"""Manages audio playback"""

import pygame
import os

class AudioManager:
    """Handles music and sound effects"""
    
    def __init__(self):
        self.music_volume = 0.5
        self.sfx_volume = 0.7
        pygame.mixer.music.set_volume(self.music_volume)
        
        # Preload sounds
        self._load_sounds()
        
    def _load_sounds(self):
        """Load all sound effects"""
        self.correct_sfx = pygame.mixer.Sound("assets/sounds/correct.wav")
        self.incorrect_sfx = pygame.mixer.Sound("assets/sounds/incorrect.wav")
        # ... other sounds
        
    def play_music(self, music_file: str, loops: int = -1):
        """Play background music"""
        pygame.mixer.music.load(music_file)
        pygame.mixer.music.play(loops)
        
    def stop_music(self):
        """Stop music"""
        pygame.mixer.music.stop()
        
    def play_sfx(self, sound: pygame.mixer.Sound):
        """Play sound effect"""
        sound.set_volume(self.sfx_volume)
        sound.play()
        
    def set_music_volume(self, volume: float):
        """Set music volume (0.0-1.0)"""
        self.music_volume = max(0.0, min(1.0, volume))
        pygame.mixer.music.set_volume(self.music_volume)
        
    def set_sfx_volume(self, volume: float):
        """Set SFX volume (0.0-1.0)"""
        self.sfx_volume = max(0.0, min(1.0, volume))
```
