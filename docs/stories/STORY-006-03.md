## Overview

Implement closed captions for all audio content in the game, including Captain Cosmos voice lines, sound effects descriptions, and environmental audio cues. This ensures students who are deaf or hard of hearing, or those in sound-sensitive environments, can fully experience the game.

## User Story

As a student who is hard of hearing or plays in a quiet environment, I want to see text captions for all audio so that I don't miss important game information.

# src/components/caption_manager.py
from typing import Optional, Deque
from collections import deque
from dataclasses import dataclass
import time

@dataclass
class Caption:
    """A single caption message"""
    text: str
    duration: float = 3.0  # seconds to display
    speaker: Optional[str] = None
    caption_id: str = ""
    timestamp: float = 0.0
    
class CaptionManager:
    """Manages caption display and queue"""
    
    def __init__(self, caption_display):
        self.display = caption_display
        self.caption_queue: Deque[Caption] = deque()
        self.current_caption: Optional[Caption] = None
        self.enabled = True
        self.expires_at = 0.0
        self.caption_history = []
        
    def show_caption(self, text: str, duration: float = 3.0,
                    speaker: Optional[str] = None, 
                    caption_id: str = "") -> None:
        """Show a caption immediately"""
        if not self.enabled:
            return
            
        caption = Caption(
            text=text,
            duration=duration,
            speaker=speaker,
            caption_id=caption_id,
            timestamp=time.time()
        )
        
        self.current_caption = caption
        self.expires_at = time.time() + duration
        self.caption_history.append(caption)
        self.display.show_caption(caption)
        
    def queue_caption(self, caption: Caption) -> None:
        """Add caption to queue for later display"""
        self.caption_queue.append(caption)
        
    def update(self, delta_time: float) -> None:
        """Update caption state (call every frame)"""
        if not self.enabled:
            return
            
        # Check if current caption expired
        if self.current_caption and time.time() >= self.expires_at:
            self.current_caption = None
            self.display.hide_caption()
            
        # Show next caption if queue has items
        if not self.current_caption and self.caption_queue:
            next_caption = self.caption_queue.popleft()
            self.show_caption(
                text=next_caption.text,
                duration=next_caption.duration,
                speaker=next_caption.speaker,
                caption_id=next_caption.caption_id
            )
            
    def clear(self) -> None:
        """Clear all captions"""
        self.caption_queue.clear()
        self.current_caption = None
        self.display.hide_caption()
        
    def set_enabled(self, enabled: bool) -> None:
        """Enable or disable captions"""
        self.enabled = enabled
        if not enabled:
            self.clear()

# src/ui/caption_display.py
import pygame

class CaptionDisplay:
    """Renders captions on screen"""
    
    def __init__(self, screen: pygame.Surface, settings):
        self.screen = screen
        self.settings = settings
        self.current_caption = None
        self.font_size = settings.font_size
        self.background_color = (0, 0, 0, 180)  # Semi-transparent black
        self.text_color = (255, 255, 255)
        self.speaker_color = (255, 215, 0)  # Gold for speaker names
        
        self.font = pygame.font.Font(None, self.font_size)
        self.speaker_font = pygame.font.Font(None, self.font_size - 8)
        
        # Position: bottom center of screen
        self posisi_x = screen.get_width() // 2
        self.position_y = screen.get_height() - 100
        
    def show_caption(self, caption) -> None:
        """Store caption for rendering"""
        self.current_caption = caption
        
    def hide_caption(self) -> None:
        """Hide current caption"""
        self.current_caption = None
        
    def update_settings(self, settings) -> None:
        """Update display settings"""
        self.settings = settings
        self.font_size = settings.font_size
        self.background_color = settings.background_color
        self.text_color = settings.text_color
        self.font = pygame.font.Font(None, self.font_size)
        
    def render(self) -> None:
        """Draw caption on screen"""
        if not self.current_caption:
            return
            
        # Calculate text dimensions
        text_surface = self.font.render(self.current_caption.text, True, self.text_color)
        text_rect = text_surface.get_rect()
        
        # Add speaker if present
        if self.current_caption.speaker:
            speaker_surface = self.speaker_font.render(
                f"{self.current_caption.speaker}:", True, self.speaker_color
            )
            speaker_width = speaker_surface.get_width()
            total_width = max(speaker_width, text_surface.get_width())
        else:
            total_width = text_surface.get_width()
            
        # Calculate background dimensions with padding
        padding = 20
        bg_width = total_width + padding * 2
        bg_height = self.font_size + padding
        
        # Create background surface
        bg_surface = pygame.Surface((bg_width, bg_height), pygame.SRCALPHA)
        bg_surface.fill(self.background_color)
        
        # Center background
        bg_rect = bg_surface.get_rect(center=(self.posisi_x, self.position_y))
        self.screen.blit(bg_surface, bg_rect)
        
        # Draw speaker name (if present)
        if self.current_caption.speaker:
            speaker_rect = speaker_surface.get_rect(
                left=bg_rect.left + padding, top=bg_rect.top + 5
            )
            self.screen.blit(speaker_surface, speaker_rect)
            text_offset_y = 5 + self.speaker_font.get_height()
        else:
            text_offset_y = padding // 2
            
        # Draw caption text
        text_rect = text_surface.get_rect(
            left=bg_rect.left + padding, 
            top=bg_rect.top + text_offset_y
        )
        self.screen.blit(text_surface, text_rect)

# src/components/caption_settings.py
from dataclasses import dataclass
from typing import Tuple

@dataclass
class CaptionSettings:
    """Caption appearance and behavior settings"""
    enabled: bool = True
    font_size: int = 28
    position: str = "bottom"  # "bottom", "middle"
    background_color: Tuple[int, int, int, int] = (0, 0, 0, 180)
    text_color: Tuple[int, int, int] = (255, 255, 255)
    speaker_color: Tuple[int, int, int] = (255, 215, 0)
    duration: float = 3.0
    high_contrast: bool = False
    
    def to_dict(self) -> dict:
        return {
            'enabled': self.enabled,
            'font_size': self.font_size,
            'position': self.position,
            'background_color': self.background_color,
            'text_color': self.text_color,
            'speaker_color': self.speaker_color,
            'duration': self.duration,
            'high_contrast': self.high_contrast
        }
    
    @classmethod
    def from_dict(cls, data: dict) -> 'CaptionSettings':
        return cls(
            enabled=data.get('enabled', True),
            font_size=data.get('font_size', 28),
            position=data.get('position', 'bottom'),
            background_color=tuple(data.get('background_color', (0, 0, 0, 180))),
            text_color=tuple(data.get('text_color', (255, 255, 255))),
            speaker_color=tuple(data.get('speaker_color', (255, 215, 0))),
            duration=data.get('duration', 3.0),
            high_contrast=data.get('high_contrast', False)
        )

## Notes & Considerations

### Edge Cases

- **Rapid Captions**: Multiple captions in quick succession - queue system handles this
- **Long Captions**: Text exceeding screen width - wrap or truncate with "..."
- **Short Captions**: Very short text (<1s reading time) - enforce minimum duration
- **Overlapping Audio**: If new audio starts before caption expires - truncate old caption

### Implementation Tips

1. **Synchronous with Audio**: Use audio player callbacks to trigger captions
2. **Fade Animations**: Smooth fade-in/fade-out for professional look
3. **Text Wrapping**: Handle long captions that exceed screen width
4. **Performance**: Use sprite caching for background surfaces

### Accessibility Best Practices

- **Reading Speed**: Average reading speed is 200-250 words/minute
- **Minimum Duration**: 2 seconds for very short captions
- **Line Length**: Maximum 32 characters per line recommended
- **Line Count**: Maximum 2 lines of text

### WCAG 2.1 AA Caption Guidelines

1. **Content**: Captions must include all significant dialogue and sound effects
2. **Synchronization**: Appear at appropriate times, displayed long enough to read
3. **Placement**: Don't block important visual content
4. **Legibility**: Clear fonts, high contrast, distinct from background
